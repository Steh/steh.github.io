public class Primzahlen {
    final static int maxInt = 100;
    static int temp;
    static int[] reihe = new int[maxInt -1];

    public static void reiheErstellen() {

        for(int i = 0; i < reihe.length; i++) {
            reihe[i] = i+2;
        }

    } // Ende reiheErstellen

    /**
    * Methode die alle Primzahlen bis zu maxInt Ausgibt
    *
    * Zuerst werden alle vielfache von 2 bis maxInt entfernt
    * Dann werden alle vielfache der nächsten Zahl, die nicht 0 sind, entfernt
    * Dies passiert so lange bis das Ende der Reihe erreicht ist
    **/
    public static void primzahlenErkennen(){
        //Schleife die durch die Komplette Reihe läuft
        for(int i = 0; i*i < reihe.length; i++) {
            //Wenn Wert in Reihe 0 dann nächster Durchlauf
            if(reihe[i] == 0) continue;
            temp = reihe[i];

            for(int x = 2; x*temp < reihe.length; x++) {
                reihe[x*temp - 2] = 0;
            }

        }
    }

    public static void reiheAusgeben(){
        System.out.print("Ausgabe der Reihe:nt");

        for(int i = 0; i < reihe.length; i++) {
            if(reihe[i] == 0) continue;
            System.out.print(reihe[i] + " ");
        }
        System.out.println();
    }

    public static void main(String[] args) {
            reiheErstellen();
            reiheAusgeben();
            primzahlenErkennen();
            reiheAusgeben();

    }


}
