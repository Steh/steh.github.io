## Grundsätzliche Bash Kommandos für Git
	# Konfiguration ausgeben
	git config --list
	
	# Konfiguration anpassen
	git config --global user.name <name>
	git config --global user.email <email>
	
	# Änderungen veröffentlichen
	git add -A
	git commit
	git push
